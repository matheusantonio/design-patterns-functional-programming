\begin{lstlisting}[caption={Bridge Orientado a Objetos},label=oobridge]

abstract class Window{
	var imp : WindowImp;

	def DrawText();
	def DrawRect();
}

class WindowImp(){
	def DevDrawText();
	def DevDrawLine();
}

class IconWindow() extends Window {
	def DrawBorder() {
		DrawRect();
		DrawText();
	}
}

class TransientWindow() extends Window {
	def DrawCloseBox() {
		DrawRect();
	}
}

class XWindowImp() extends WindowImp {
	def DevDrawText() {
		XDrawString();
	}

	def DevDrawLine() {
		XDrawLine();
	}
}

class PMWindowImp() extends WindowImp {
	def DevDrawLine() {
		PMDrawLine();
	}

	def DevDrawText() {
		PMDrawString();
	}
}

\end{lstlisting}
